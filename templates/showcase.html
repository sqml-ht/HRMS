 vw_employee_details

Virtual table joining employees → departments → designations → employees (manager self-join). Shows all active employees with full context in one query.

CREATE OR REPLACE VIEW vw_employee_details AS SELECT e.emp_id, e.first_name || ' ' || e.last_name AS full_name, e.email, e.salary, e.hire_date, e.status, d.dept_name, dg.title AS designation, dg.grade, m.first_name || ' ' || m.last_name AS manager_name FROM employees e LEFT JOIN departments d ON e.dept_id = d.dept_id LEFT JOIN designations dg ON e.desig_id = dg.desig_id LEFT JOIN employees m ON e.manager_id = m.emp_id -- SELF JOIN WHERE e.status = 'ACTIVE';

 vw_attendance_summary

Uses CASE WHEN · COUNT · SUM · ROUND · NULLIF to compute per-employee attendance stats. Demonstrates conditional aggregation.

SELECT e.emp_id, e.first_name || ' ' || e.last_name AS full_name, COUNT(CASE WHEN a.status = 'PRESENT' THEN 1 END) AS present_days, COUNT(CASE WHEN a.status = 'ABSENT' THEN 1 END) AS absent_days, NVL(SUM(a.overtime_hr), 0) AS total_overtime, ROUND( COUNT(CASE WHEN a.status='PRESENT' THEN 1 END) * 100.0 / NULLIF(COUNT(a.att_id), 0), 1) AS attendance_pct FROM employees e LEFT JOIN attendance a ON e.emp_id = a.emp_id LEFT JOIN departments d ON e.dept_id = d.dept_id WHERE e.status = 'ACTIVE' GROUP BY e.emp_id, e.first_name, e.last_name, d.dept_name;

 vw_dept_payroll_summary

Department-level payroll rollup using INNER JOIN · GROUP BY · SUM across payroll, employees, and departments.

SELECT d.dept_name, p.pay_month, p.pay_year, COUNT(p.pay_id) AS employee_count, SUM(p.basic_pay) AS total_basic, SUM(p.bonus) AS total_bonus, SUM(p.deductions) AS total_deductions, SUM(p.net_pay) AS total_net_pay FROM payroll p JOIN employees e ON p.emp_id = e.emp_id JOIN departments d ON e.dept_id = d.dept_id GROUP BY d.dept_name, p.pay_month, p.pay_year ORDER BY p.pay_year DESC, p.pay_month, d.dept_name;

 sp_generate_payroll

Iterates all active employees with a cursor FOR loop, calculates bonus from overtime hours (500/hr), applies 5% deductions + 10% tax, inserts payroll with COMMIT. Skips already-generated records.

CREATE OR REPLACE PROCEDURE sp_generate_payroll ( p_month IN VARCHAR2, p_year IN NUMBER ) AS v_overtime NUMBER; v_bonus NUMBER; ... BEGIN FOR emp IN (SELECT emp_id, salary FROM employees WHERE status='ACTIVE') LOOP -- Skip if already generated v_bonus := v_overtime * 500; v_deductions := v_basic * 0.05; v_tax := v_basic * 0.10; v_net := v_basic + v_bonus - v_deductions - v_tax; INSERT INTO payroll ...; END LOOP; COMMIT; END;

 sp_dept_salary_raise

Uses SAVEPOINT · COMMIT · ROLLBACK to safely apply a % salary raise to a department. Triggers trg_salary_history automatically on each row updated.

CREATE OR REPLACE PROCEDURE sp_dept_salary_raise ( p_dept_id IN NUMBER, p_raise_pct IN NUMBER ) AS BEGIN SAVEPOINT before_raise; UPDATE employees SET salary = salary * (1 + p_raise_pct / 100) WHERE dept_id = p_dept_id AND status = 'ACTIVE'; COMMIT; EXCEPTION WHEN OTHERS THEN ROLLBACK TO before_raise; END;

 ACID Transaction Demo

ACID Properties
Atomicity — all or nothing
Consistency — valid state always
Isolation — concurrent transactions
Durability — committed = permanent
What This Demo Does
Sets a SAVEPOINT → modifies salary → reads the in-transaction value → then either COMMITs (durable change) or ROLLBACKs to savepoint (atomicity).
BEGIN SAVEPOINT acid_demo_sp; -- mark restore point UPDATE employees SET salary = salary + :amount WHERE emp_id = :emp_id; -- dirty change SELECT salary FROM employees WHERE emp_id = :emp_id; -- read within txn -- Choose path: COMMIT; -- Durability: change persists -- OR -- ROLLBACK TO SAVEPOINT acid_demo_sp; -- Atomicity: undo END;
